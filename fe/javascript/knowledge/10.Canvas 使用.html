function createPolygon(ctx, n, dx, dy, size) {
  ctx.beginPath()
  var degree = (2 * Math.PI) / n
  for (var i = 0; i < n; i++) {
    var x = Math.cos(i * degree)
    var y = Math.sin(i * degree)
    ctx.lineTo(x * size + dx, y * size + dy)
  }
  ctx.closePath()
}

  }, 200)
}
